import { X } from 'lucide-react';
import { useGraphStore } from '../../store/useGraphStore';
 
export const NodeInspector = () => {
  const { selectedNode, setSelectedNode } = useGraphStore();
 
  if (!selectedNode) return null;
 
  // Using real node data if available
  const nodeId = selectedNode?.id || 'Unknown';
  const nodeLabel = selectedNode?.label || 'Unknown';
  const properties = selectedNode?.properties || {};
 
  const nodeData = {
    nombre: nodeId,
    tipo: nodeLabel,
    centralidad: properties.centrality_score?.toFixed(2) || properties.PageRank?.toFixed(2) || 'N/A',
    influencia: properties.influence_score?.toFixed(2) || 'N/A',
    dependencias: `${properties.degree || 0} connections`,
    impacto: properties.centrality_score > 0.7 ? 'Alto' : 'Normal',
    gobernanza: properties.governance_status || 'N/A',
    estado: properties.status || 'Activo',
    riesgo: properties.risk_level || 'Bajo',
    recomendaciones: properties.recommendations || []
  };
 
  return (
    <div className="w-80 glassmorphism rounded-xl flex flex-col border-t-4 border-blue-500 overflow-hidden">
      <div className="p-4 border-b border-white/10 flex justify-between items-center bg-white/5">
        <h3 className="text-lg font-bold text-white">Node Inspector</h3>
        <button onClick={() => setSelectedNode(null)} className="text-gray-400 hover:text-white transition-colors">
          <X size={20} />
        </button>
      </div>
      
      <div className="p-6 overflow-y-auto space-y-4 flex-1">
        <div className="space-y-1">
          <span className="text-gray-400 text-xs uppercase font-semibold">Nombre</span>
          <p className="text-gray-200 text-sm break-all">{nodeData.nombre}</p>
        </div>
        
        <div className="grid grid-cols-2 gap-4">
          <div className="space-y-1">
            <span className="text-gray-400 text-xs uppercase font-semibold">Tipo</span>
            <p className="text-gray-200 text-sm">{nodeData.tipo}</p>
          </div>
          <div className="space-y-1">
            <span className="text-gray-400 text-xs uppercase font-semibold">Estado</span>
            <p className="text-emerald-400 text-sm">{nodeData.estado}</p>
          </div>
        </div>
 
        <div className="grid grid-cols-2 gap-4">
          <div className="space-y-1">
            <span className="text-gray-400 text-xs uppercase font-semibold">Centralidad</span>
            <p className="text-blue-400 text-sm">{nodeData.centralidad}</p>
          </div>
          <div className="space-y-1">
            <span className="text-gray-400 text-xs uppercase font-semibold">Influencia</span>
            <p className="text-purple-400 text-sm font-bold">{nodeData.influencia}</p>
          </div>
        </div>
 
        <div className="space-y-1">
          <span className="text-gray-400 text-xs uppercase font-semibold">Dependencias</span>
          <p className="text-gray-200 text-sm">{nodeData.dependencias}</p>
        </div>
 
        <div className="grid grid-cols-2 gap-4">
          <div className="space-y-1">
            <span className="text-gray-400 text-xs uppercase font-semibold">Impacto</span>
            <p className="text-red-400 text-sm">{nodeData.impacto}</p>
          </div>
          <div className="space-y-1">
            <span className="text-gray-400 text-xs uppercase font-semibold">Riesgo</span>
            <p className="text-amber-400 text-sm">{nodeData.riesgo}</p>
          </div>
        </div>
 
        <div className="space-y-1">
          <span className="text-gray-400 text-xs uppercase font-semibold">Gobernanza</span>
          <p className="text-gray-200 text-sm">{nodeData.gobernanza}</p>
        </div>
 
        {nodeData.recomendaciones.length > 0 && (
          <div className="space-y-2 mt-4 pt-4 border-t border-white/10">
            <span className="text-gray-400 text-xs uppercase font-semibold">Recomendaciones</span>
            <ul className="list-disc pl-4 space-y-1">
              {nodeData.recomendaciones.map((rec: string, idx: number) => (
                <li key={idx} className="text-xs text-orange-300">{rec}</li>
              ))}
            </ul>
          </div>
        )}
      </div>
    </div>
  );
};
